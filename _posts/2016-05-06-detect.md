### 1.PC平台判断
通过使用：`navigator.platform`，判断主要的PC平台。

```js
/**
 * [platform pc平台检测]
 * @param  {[type]} ua [description]
 * @return {[type]}    [description]
 */
function platform(ua){
  var os ={
    win : false,
    mac : false,
    xll : false
  } , ua = navigator.platform;
  //检测平台
  os.win = ua.match("Win");
  os.mac = ua.match("Mac");
  os.xll = ua.match("X11") || ua.match("Linux");
  
  return os;
};
var os = platform();
```
